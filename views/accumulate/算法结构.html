# 2. 计数二进制子串

输入: "00110011"
输出: 6
解释: 有6个子串具有相同数量的连续1和0：“0011”，“01”，“1100”，“10”，“0011” 和 “01”。

请注意，一些重复出现的子串要计算它们出现的次数。

另外，“00110011”不是有效的子串，因为所有的0（和1）没有组合在一起。
#### 解:
export default (str) => {
  // 建立数据结构，堆栈，保存数据
  let r = []
  // 给定任意子输入都返回第一个符合条件的子串
  let match = (str) => {
    let j = str.match(/^(0+|1+)/)[0]
    let o = (j[0] ^ 1).toString().repeat(j.length)
    let reg = new RegExp(`^(${j}${o})`)
    if (reg.test(str)) {
      return RegExp.$1
    } else {
      return ''
    }
  }
  // 通过for循环控制程序运行的流程
  for (let i = 0, len = str.length - 1; i < len; i++) {
    let sub = match(str.slice(i))
    if (sub) {
      r.push(sub)
    }
  }
  return r
}

# 数组

# 1. 电话号码的字母组合(组合运算)

给定一个仅包含数字 2-9 的字符串，返回所有它能表示的字母组合。

给出数字到字母的映射如下（与电话按键相同）。注意 1 不对应任何字母。
输入："23"
输出：["ad", "ae", "af", "bd", "be", "bf", "cd", "ce", "cf"]

# 解:

思路:

建立数组映射, 将数组下标与号码字母相对应 号码有可能会输入多个, 所以输入个数是不确定的, 将运算拆解, 两两相算, 永远保持数组第一项与第二项组合运算

export default (str) => {
  // 首先建立数组映射
  let map = ['', 1, 'abc', 'def', 'ghi', 'jkl', 'mno', 'pqrs', 'tuv', 'wxyz']
  // 将索引转换为数组 '234' =>> ['2', '3', '4']
  let numArr = str.split('')
  let code = numArr.reduce((p, c) => {
    map[c] && p.push(map[c])
    return p
  }, [])
  let func = (arr) => {
    let temp = []
    for (let i = 0, il = arr[0].length; i < il; i++) {
      for (let j = 0, jl = arr[1].length; j < jl; j++) {
        temp.push(`${arr[0][i]}${arr[1][j]}`)
      }
    }
    arr.splice(0, 2, temp)
    if (arr.length > 1) {
      func(arr)
    } else {
      return temp
    }
    return arr[0]
  }
  return func(code)
}

# 2.卡片分组

给定一副牌，每张牌上都写着一个整数。

此时，你需要选定一个数字 X，使我们可以将整副牌按下述规则分成 1 组或更多组：

    每组都有 X 张牌。
    组内所有的牌上都写着相同的整数。

仅当你可选的 X >= 2 时返回 true
输入：[1,2,3,4,4,3,2,1]
输出：true
解释：可行的分组是 [1,1]，[2,2]，[3,3]，[4,4]

# 解:

export default (arr) => {
  arr.sort()
  let min = Number.MAX_SAFE_INTEGER
  let dir = []
  for (let i = 0, len = arr.length, temp = []; i < len; i++) {
    temp.push(arr[i])
    for (let j = i + 1; j < len - 1; j++) {
      if (arr[i] === arr[j]) {
        temp.push(arr[j])
      } else {
        if (temp.length < min) {
          min = temp.length
        }
        dir.push([].concat(temp))
        temp.length = 0
        i = j
        break
      }
    }
  }
  return dir.every(item => item.length >= 2 && item.length % min === 0)
}

# 3. 种花问题

假设你有一个很长的花坛，一部分地块种植了花，另一部分却没有。可是，花卉不能种植在相邻的地块上，它们会争夺水源，两者都会死去。

给定一个花坛（表示为一个数组包含0和1，其中0表示没种植花，1表示种植了花），和一个数 n 。能否在不打破种植规则的情况下种入 n 朵花？能则返回True，不能则返回False。

示例 1:

输入: flowerbed = [1,0,0,0,1], n = 1
输出: True

示例 2:

输入: flowerbed = [1,0,0,0,1], n = 2
输出: False

# 解:

export default (arr, n) => {
  let num = 0
  for (let i = 0, il = arr.length; i < il; i++) {
    if (i === 0 && arr[1] === 0) {
      num++
      i += 1
    } else if (arr[i - 1] === 0 && arr[i + 1] === 0) {
      num++
      i += 1
    }
  }
  return num >= n
}
